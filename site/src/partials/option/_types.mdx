import { Code } from 'astro:components';

Override your [dependencyTypes](/syncpack/config/dependency-types) configuration to run <code>{`syncpack ${props.command}`}</code> on only eg. `peerDependencies` in every package in your monorepo.

<Code
  code={`# only include peerDependencies
syncpack ${props.command} --types 'peer'
# only include dependencies and devDependencies
syncpack ${props.command} --types 'prod,dev'
# exclude local package.json versions
syncpack ${props.command} --types '!local'`}
  lang="bash"
/>

import { Code } from 'astro:components';

Override your [indent](/syncpack/config/indent) configuration, which is 2 spaces by default.

<Code
  code={`# four spaces
syncpack ${props.command} --indent "    "
# one tab
syncpack ${props.command} --indent '\t'
# \\t is converted into a tab
syncpack ${props.command} --indent '\\t'`}
  lang="bash"

/>

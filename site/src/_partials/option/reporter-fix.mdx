Set the output format. Defaults to `pretty` which prints human-readable output with colours and formatting. Use `json` to output one line of newline-delimited JSON (NDJSON) per fixable instance — the same shape as the [json](COMMAND_JSON) command.

Each line has the type [`JsonOutput`](https://github.com/JamieMason/syncpack/blob/14.1.0/npm/syncpack.ts#L174-L194):

```jsonc
// This JSON has been formatted but appears on a single line in syncpack
{
  "dependency": "kleur",
  "dependencyGroup": "kleur",
  "dependencyType": "prod",
  "package": "apps/billing/package.json",
  "property": [
    "dependencies"
  ],
  "strategy": "versionsByName",
  "versionGroup": "HighestSemver",
  "preferredSemverRange": "",
  "statusCode": "DiffersToHighestSemver",
  "statusType": "Fixable",
  "actual": {
    "raw": "4.1.4",
    "type": "exact"
  },
  "expected": {
    "raw": "4.1.5",
    "type": "exact"
  }
}
```

<Code
code={`# Default human-readable output
syncpack ${props.command} --reporter pretty
# Machine-readable NDJSON output
syncpack ${props.command} --reporter json
# Pipe to jq to filter fixable instances
syncpack ${props.command} --reporter json | jq 'select(.statusCode == "DiffersToHighestSemver")'
# Filter by statusType to get all fixable instances
syncpack ${props.command} --reporter json | jq 'select(.statusType == "Fixable")'
# Get only unfixable instances that need manual intervention
syncpack ${props.command} --reporter json | jq 'select(.statusType == "Unfixable")'
# Get suspect instances (possible misconfigurations)
syncpack ${props.command} --reporter json | jq 'select(.statusType == "Suspect")'`}
lang="bash"
/>
